代码示例

<!-- 水平分隔线 -->
<span class="separator separator-horizontal separator-size-4"></span>

<!-- 垂直分隔线 -->
<span class="separator separator-vertical separator-size-2"></span>

<!-- 不同尺寸 -->
<span class="separator separator-horizontal separator-size-1"></span> <!-- Short -->
<span class="separator separator-horizontal separator-size-2"></span> <!-- Medium -->
<span class="separator separator-horizontal separator-size-3"></span> <!-- Long -->
<span class="separator separator-horizontal separator-size-4"></span> <!-- Full -->

<!-- 自定义颜色 (使用 data-accent-color) -->
<div data-accent-color="blue">
  <span class="separator separator-horizontal separator-size-4"></span>
</div>

<!-- 导航菜单示例 -->

  <a href="#">Contact</a>
</div>
