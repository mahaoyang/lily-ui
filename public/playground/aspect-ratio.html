代码示例

<!-- 使用预设宽高比 -->
<div class="aspect-ratio aspect-ratio-16-9">
  <img src="image.jpg" alt="...">
</div>

<!-- 1:1 正方形 -->
<div class="aspect-ratio aspect-ratio-1-1">
  <img src="image.jpg" alt="...">
</div>

<!-- 4:3 标准 -->
<div class="aspect-ratio aspect-ratio-4-3">
  <img src="image.jpg" alt="...">
</div>

<!-- 3:2 照片 -->
<div class="aspect-ratio aspect-ratio-3-2">
  <img src="image.jpg" alt="...">
</div>

<!-- 21:9 超宽屏 -->
<div class="aspect-ratio aspect-ratio-21-9">
  <img src="image.jpg" alt="...">
</div>

<!-- 9:16 竖屏/故事 -->
<div class="aspect-ratio aspect-ratio-9-16">
  <img src="image.jpg" alt="...">
</div>

<!-- 自定义宽高比 -->
<div class="aspect-ratio" style="--aspect-ratio: 2.35;">
  <img src="image.jpg" alt="...">
</div>

<!-- 自定义黄金比例 -->
<div class="aspect-ratio" style="--aspect-ratio: 1.618;">
  <img src="image.jpg" alt="...">
</div>

<!-- Object fit 变体 -->
<div class="aspect-ratio aspect-ratio-1-1 aspect-ratio-cover">
  <img src="image.jpg" alt="...">
</div>

<div class="aspect-ratio aspect-ratio-1-1 aspect-ratio-contain">
  <img src="image.jpg" alt="...">
</div>
