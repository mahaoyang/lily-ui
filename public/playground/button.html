代码示例

<!-- 基本用法 -->
<button class="button button-size-2 button-variant-solid">Button</button>

<!-- 不同尺寸 -->
<button class="button button-size-1 button-variant-solid">Small</button>
<button class="button button-size-2 button-variant-solid">Default</button>
<button class="button button-size-3 button-variant-solid">Large</button>
<button class="button button-size-4 button-variant-solid">XLarge</button>

<!-- 不同变体 -->
<button class="button button-size-2 button-variant-classic">Classic</button>
<button class="button button-size-2 button-variant-solid">Solid</button>
<button class="button button-size-2 button-variant-soft">Soft</button>
<button class="button button-size-2 button-variant-surface">Surface</button>
<button class="button button-size-2 button-variant-outline">Outline</button>
<button class="button button-size-2 button-variant-ghost">Ghost</button>

<!-- 禁用状态 -->
<button class="button button-size-2 button-variant-solid" disabled>
  Disabled
</button>

<!-- 加载状态 -->
<button class="button button-size-2 button-variant-solid" data-loading>
  <span class="button-content">Loading</span>
  <span class="button-spinner spinner spinner-size-2">
    <!-- 8个 spinner-leaf -->
  </span>
</button>

<!-- 带图标 -->
<button class="button button-size-2 button-variant-solid">
  <svg>...</svg>
  Next
</button>

<!-- 自定义颜色 -->
<div data-accent-color="red">
  <button class="button button-size-2 button-variant-solid">Delete</button>
</div>
