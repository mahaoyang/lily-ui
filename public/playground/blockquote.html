代码示例

<!-- 基本用法 -->
<blockquote class="blockquote blockquote-size-3">
  Your quote text here.
</blockquote>

<!-- 不同尺寸 (1-9) -->
<blockquote class="blockquote blockquote-size-1">Small quote</blockquote>
<blockquote class="blockquote blockquote-size-3">Default quote</blockquote>
<blockquote class="blockquote blockquote-size-6">Large quote</blockquote>

<!-- 字重变体 -->
<blockquote class="blockquote blockquote-size-3 blockquote-weight-light">
  Light weight quote
</blockquote>
<blockquote class="blockquote blockquote-size-3 blockquote-weight-bold">
  Bold quote
</blockquote>

<!-- 自定义颜色 -->
<blockquote class="blockquote blockquote-size-3" data-accent-color="blue">
  Blue accent quote
</blockquote>

<!-- 高对比度 -->
<blockquote class="blockquote blockquote-size-3 blockquote-high-contrast"
  data-accent-color="iris">
  High contrast quote
</blockquote>

<!-- 带引用来源 -->
<blockquote class="blockquote blockquote-size-4">
  Stay hungry, stay foolish.
</blockquote>
<cite class="text-sm text-sand-11 not-italic">
  — Steve Jobs
</cite>

<!-- 嵌套引用 -->
<blockquote class="blockquote blockquote-size-3" data-accent-color="iris">
  Outer quote
  <blockquote class="blockquote blockquote-size-2" data-accent-color="blue">
    Nested quote
  </blockquote>
</blockquote>
