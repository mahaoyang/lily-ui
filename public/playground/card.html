代码示例

<!-- 基本用法 -->
<div class="card card-size-2 card-surface">
  <p>Card content</p>
</div>

<!-- 不同尺寸 -->
<div class="card card-size-1 card-surface">Compact</div>
<div class="card card-size-3 card-surface">Medium</div>
<div class="card card-size-5 card-surface">Extra Large</div>

<!-- 不同变体 -->
<div class="card card-size-3 card-surface">Surface</div>
<div class="card card-size-3 card-classic">Classic</div>
<div class="card card-size-3 card-ghost">Ghost</div>

<!-- 可点击的卡片 -->
<a href="#" class="card card-size-3 card-surface">
  Clickable card
</a>

<button class="card card-size-3 card-classic">
  Interactive card
</button>

<!-- 带 Inset 的卡片 -->
<div class="card card-size-3 card-surface">
  <div class="card-inset">
    <img src="image.jpg" alt="Cover" />
  </div>
  <p>Content below image</p>
</div>
