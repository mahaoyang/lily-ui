代码示例

<!-- 基本用法 -->
<div x-data="switchMachine({ id: 'my-switch' })">
  <button x-bind="switchProps()" type="button"
    class="switch switch-size-2 switch-surface"
    :data-state="checked ? 'checked' : 'unchecked'">
    <span class="switch-thumb"></span>
  </button>
  <input x-bind="inputProps()" class="hidden" />
</div>

<!-- 尺寸变体 -->

<!-- 高对比度 -->
