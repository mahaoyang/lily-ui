代码示例

<!-- 基本用法（仅 Fallback）-->
<span class="avatar-root avatar-size-3 avatar-variant-soft">
  <span class="avatar-fallback" data-letters="1">A</span>
</span>

<!-- 带图片（自动 Fallback）-->
<span class="avatar-root avatar-size-3 avatar-variant-soft">
  <img class="avatar-image" src="avatar.jpg" alt="User">
  <span class="avatar-fallback" data-avatar-fallback data-letters="2">AB</span>
</span>

<!-- JavaScript 初始化 -->
<script type="module">
  import { createAvatar } from "/public/machines/avatar.js";

  const avatarEl = document.querySelector('.avatar-root');
  const machine = createAvatar(avatarEl);

  // 获取当前状态
  console.log(machine.getStatus()); // 'idle' | 'loading' | 'loaded' | 'error'

  // 清理
  machine.destroy();
</script>

<!-- 不同尺寸 -->
<span class="avatar-root avatar-size-1 avatar-variant-soft">...</span>
<span class="avatar-root avatar-size-5 avatar-variant-soft">...</span>
<span class="avatar-root avatar-size-9 avatar-variant-soft">...</span>

<!-- 不同变体 -->
<span class="avatar-root avatar-size-3 avatar-variant-solid">...</span>
<span class="avatar-root avatar-size-3 avatar-variant-soft">...</span>

<!-- 高对比度 -->
<span class="avatar-root avatar-size-3 avatar-variant-soft avatar-high-contrast">
  <span class="avatar-fallback" data-letters="1">H</span>
</span>

<!-- 自定义颜色 -->
<div data-accent-color="red">
  <span class="avatar-root avatar-size-3 avatar-variant-solid">
    <span class="avatar-fallback" data-letters="1">R</span>
  </span>
</div>

<!-- 自定义圆角 -->
<span class="avatar-root avatar-size-3 avatar-variant-soft" data-radius="full">
  <span class="avatar-fallback" data-letters="1">F</span>
</span>
