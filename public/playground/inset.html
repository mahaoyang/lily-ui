<!-- 基础边到边内容 -->
<div class="rounded-lg border bg-white p-6" style="
  --inset-padding-top: 1.5rem;
  --inset-padding-right: 1.5rem;
  --inset-padding-bottom: 1.5rem;
  --inset-padding-left: 1.5rem;
  --inset-border-radius: 0.5rem;
  --inset-border-width: 1px;
">
  <h3>Card Title</h3>
  <div class="inset inset-all inset-clip-border-box">
    <img src="..." alt="..." class="w-full h-48 object-cover">
  </div>
</div>

<!-- 顶部边到边 -->
<div class="rounded-lg border bg-white p-6" style="
  --inset-padding-top: 1.5rem;
  --inset-padding-right: 1.5rem;
  --inset-padding-bottom: 1.5rem;
  --inset-padding-left: 1.5rem;
  --inset-border-radius: 0.5rem;
  --inset-border-width: 1px;
">
  <div class="inset inset-top inset-clip-border-box">
    <img src="..." alt="..." class="w-full h-40 object-cover">
  </div>
  <h3 class="mt-4">Title</h3>
  <p>Content...</p>
</div>

<!-- 水平边到边（左右） -->
<div class="rounded-lg border bg-white p-6" style="
  --inset-padding-top: 1.5rem;
  --inset-padding-right: 1.5rem;
  --inset-padding-bottom: 1.5rem;
  --inset-padding-left: 1.5rem;
  --inset-border-radius: 0.5rem;
  --inset-border-width: 1px;
">
  <h3>Section 1</h3>
  <div class="inset inset-x inset-clip-border-box">
    <span class="separator separator-horizontal separator-size-4"></span>
  </div>
  <h3>Section 2</h3>
</div>

<!-- CSS 变量说明 -->
--inset-padding-top:     父容器的上内边距
--inset-padding-right:   父容器的右内边距
--inset-padding-bottom:  父容器的下内边距
--inset-padding-left:    父容器的左内边距
--inset-border-radius:   父容器的圆角半径
--inset-border-width:    父容器的边框宽度

<!-- 侧边选项 -->
.inset-all      全部边（上下左右）
.inset-x        水平边（左右）
.inset-y        垂直边（上下）
.inset-top      顶部
.inset-bottom   底部
.inset-left     左侧
.inset-right    右侧

<!-- 裁剪模式 -->
.inset-clip-border-box   考虑边框宽度（默认）
.inset-clip-padding-box  包含边框在内
