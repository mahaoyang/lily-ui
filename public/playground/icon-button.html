<!-- 基本用法 -->

  <svg>...</svg>
</button>

  <svg>...</svg>
</button>

<!-- 不同变体 -->
<button class="icon-button icon-button-size-2 icon-button-variant-classic" aria-label="Icon">
  <svg>...</svg>
</button>
<button class="icon-button icon-button-size-2 icon-button-variant-solid" aria-label="Icon">
  <svg>...</svg>
</button>
<button class="icon-button icon-button-size-2 icon-button-variant-soft" aria-label="Icon">
  <svg>...</svg>

  <svg>...</svg>
</button>

  <svg>...</svg>
</button>
<button class="icon-button icon-button-size-2 icon-button-variant-ghost" aria-label="Icon">
  <svg>...</svg>
</button>

<!-- 禁用状态 -->
<button class="icon-button icon-button-size-2 icon-button-variant-solid" disabled aria-label="Icon">
  <svg>...</svg>
</button>

<!-- 加载状态 -->
<button class="icon-button icon-button-size-2 icon-button-variant-solid" data-loading aria-label="Loading">
  <svg>...</svg>
  <span class="icon-button-spinner spinner spinner-size-2">
    <!-- 8个 spinner-leaf -->
  </span>
</button>

<!-- 高对比度 -->
<button class="icon-button icon-button-size-2 icon-button-variant-solid icon-button-high-contrast" aria-label="Icon">
  <svg>...</svg>
</button>

<!-- 自定义颜色 -->
<button class="icon-button icon-button-size-2 icon-button-variant-solid" data-accent-color="red" aria-label="Delete">
  <svg>...</svg>
</button>
